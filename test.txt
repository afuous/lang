# find the nth prime number
let total
input total
let to_check = 2
let count = 0
while total > count {
	let n = 2
	let cont = 1
	while cont == 1 {
		if to_check == n {
			cont = 0
			count = count + 1
		} else {
			if to_check % n == 0 {
				cont = 0
			}
		}
		n = n + 1
	}
	to_check = to_check + 1
}
output to_check - 1
