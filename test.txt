# find the nth prime number
input total
tocheck = 2
count = 0
while total - count {
	n = 2
	cont = 1
	while cont {
		if 0 ^ (tocheck - n) { # if n equals tocheck
			cont = 0
			count = count + 1
		} else {
			if 0 ^ (tocheck % n) { # if n divides tocheck
				cont = 0
			}
		}
		n = n + 1
	}
	tocheck = tocheck + 1
}
output tocheck - 1
