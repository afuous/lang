# find the nth prime number
input total
to_check = 2
count = 0
while total > count {
	n = 2
	cont = True
	while cont {
		if to_check == n {
			cont = False
			count = count + 1
		} else {
			if to_check % n == 0 {
				cont = False
			}
		}
		n = n + 1
	}
	to_check = to_check + 1
}
output to_check - 1
