# find the nth prime number
input total
tocheck = 2
count = 0
while total > count {
	n = 2
	cont = 1 == 1
	while cont {
		if tocheck == n { # if n equals tocheck
			cont = 1 == 0
			count = count + 1
		} else {
			if tocheck % n == 0 {
				cont = 1 == 0
			}
		}
		n = n + 1
	}
	tocheck = tocheck + 1
}
output tocheck - 1
