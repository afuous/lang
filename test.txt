# find the nth prime number
input total
tocheck = 2
count = 0
while total > count {
	n = 2
	cont = True
	while cont {
		if tocheck == n {
			cont = False
			count = count + 1
		} else {
			if tocheck % n == 0 {
				cont = False
			}
		}
		n = n + 1
	}
	tocheck = tocheck + 1
}
output tocheck - 1
