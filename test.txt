# find the nth prime number
let total
input total
let to_check = 2
let count = 0
while total > count {
	let n = 2
	let cont = True
	while cont {
		if to_check == n {
			cont = False
			count = count + 1
		} else {
			if to_check % n == 0 {
				cont = False
			}
		}
		n = n + 1
	}
	to_check = to_check + 1
}
output to_check - 1
